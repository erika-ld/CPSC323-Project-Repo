[* this is comment for this sample code which
converts Fahrenheit into Celcius *]
$
function convertx (fahr integer)
{
return 5 * (fahr -32) / 9;
}
$
integer low, high, step; [* declarations *]
$ 2.00 / .12
scan (low, high, step);
while (low <= high )
{ print (low);
print (convertx (low));
low = low + step;
}
endwhile
